ExtendedKey({
  String rootPath,
  Uint8List code,
  int depth,
  BigInt i,
  Uint8List parent,
  String path,
}) {
  this.rootPath = rootPath;
  this.parent = (parent != null) ? parent : Uint8List.fromList([0,0,0,0]);
  this.code = code;
  this.depth = depth;
  this.i = i;
  this.path = path;
  assert (255 >= depth && depth >= 0, 'Depth can only be 0-255');
  if (i != null){
   assert (BigInt.zero <= i && i < BigInt.from(1 << 32), 'Invalid i : $i');
 }
}
