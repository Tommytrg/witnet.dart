Xpub child(BigInt index) {
  bool hardened = index >= BigInt.from(1 << 31);
  if (hardened)
    assert(hardened == false, 'Cannot derive a hardened key from an extended public key');
  Uint8List I = hmacSHA512(key: code, data: keyData + bigIntToBytes(index));
  Uint8List IL = I.sublist(0,32);
  Uint8List IR = I.sublist(32);
  final p = WitPrivateKey(bytes: IL).publicKey.point + publicKey.point;
  return Xpub(
      publicKey: WitPublicKey(X:p.x.toBigInteger(), Y:p.y.toBigInteger()),
      code: IR,
      depth: depth + 1,
      index: index,
      parent: fingerPrint,
      path: path + '/$index');
}
